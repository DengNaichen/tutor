\section*{Concept Overview}
Success in kinematics depends on two core skills:
\begin{enumerate}
    \item \textbf{Process Decoupling:} Breaking a continuous motion into intervals that share common "States" (e.g., the final velocity of part A is the initial velocity of part B).
    \item \textbf{Frame Transformation:} Shifting from a Lab Frame (Ground) to a Relative Frame to simplify multi-body problems.
\end{enumerate}


\section*{Problem 1: The Pursuit (Two-Body Kinematics)}
\textbf{The Scenario:} A speeder passes a parked police car at a constant velocity of $v_c = 30 \text{ m/s}$. The police officer starts to accelerate from rest at a constant rate of $a = 5 \text{ m/s}^2$ exactly $1 \text{ s}$ after the speeder passes.

\begin{itemize}
    \item[(a)] Calculate the time $t$ it takes for the police car to overtake the speeder (measured from the moment the police car starts).
    \item[(b)] Determine the maximum separation distance $D_{max}$ between the two cars before the intercept.
\end{itemize}


\section*{Problem 2: The Closest Approach (Relative Motion)}
\textbf{The Scenario:} At $t = 0$, Ship A is $10 \text{ km}$ due North of Ship B. Ship A moves South at $20 \text{ km/h}$, and Ship B moves East at $15 \text{ km/h}$. Both maintain constant velocities.

\begin{itemize}
    \item[(a)] Find the minimum distance $d_{min}$ between the ships during their journey.
\end{itemize}

\section*{Problem 3: The Interval Challenge (Process Analysis)}
\textbf{The Scenario:} An object is released from rest from the top of a tower. In its final $1.0 \text{ s}$ of fall, it covers a distance equal to the distance it covered in its first $3.0 \text{ s}$ of fall. (Assume $g = 10 \text{ m/s}^2$).

\begin{itemize}
    \item[(a)] What is the total height $H$ of the tower?
\end{itemize}
